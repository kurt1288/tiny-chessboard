tiny-chessboard.js

Basic Initialization

No config


import Chessboard from "./tiny-chessboard.min.js";

const board = new Chessboard("chessboard");

Config with position


import Chessboard from "./tiny-chessboard.min.js";

const board = new Chessboard("chessboard", {
    position: "rnbqk2r/pppp1ppp/4pn2/8/1bPP4/2N5/PP2PPPP/R1BQKBNR w KQkq - 2 4"
});
